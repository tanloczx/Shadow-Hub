-- Logic Bug CD cc ngon ca bn
local function RunStoreLogic()
    local backpack = lp.Backpack
    local char = lp.Character
    local hum = char:FindFirstChildOfClass("Humanoid")
    local saveRemote = game:GetService("ReplicatedStorage").Save.SavingEvent
    
    local item = nil
    local targetNames = {"Box", "Book", "Fruit"}
    
    for _, name in pairs(targetNames) do
        for _, v in pairs(backpack:GetChildren()) do
            if v:IsA("Tool") and v.Name:find(name) then
                item = v; break
            end
        end
        if item then break end
    end

    if item and hum then
        local itemName = item.Name
        hum:EquipTool(item)
        task.wait(0.15)
        saveRemote:FireServer(1) 
        
        local count = 0
        while char:FindFirstChild(itemName) and count < 10 do
            task.wait(0.05)
            count = count + 1
        end

        task.wait(0.1)
        saveRemote:FireServer(1) 

        local timeout = 0
        while not backpack:FindFirstChild(itemName) and timeout < 20 do
            task.wait(0.05)
            timeout = timeout + 1
        end
        
        local newItem = backpack:FindFirstChild(itemName)
        if newItem then hum:EquipTool(newItem) end
    end
end

MainBtn.MouseButton1Click:Connect(function()
    if _G.IsRunning then return end
    _G.IsRunning = true
    
    MainBtn:TweenSize(UDim2.new(0, 55, 0, 55), "Out", "Quad", 0.1)
    MainBtn.Text = "FIXING"
    
    pcall(RunStoreLogic)
    
    MainBtn:TweenSize(UDim2.new(0, 65, 0, 65), "Out", "Quad", 0.1)
    MainBtn.Text = "SHADOW"
    _G.IsRunning = false
end)
